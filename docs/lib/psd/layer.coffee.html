Every layer starts with the same set of data, and ends with a dynamic number of layer info blocks.

  parse: ->
    @parsePositionAndChannels()
    @parseBlendModes()

    extraLen = @file.readInt()
    @layerEnd = @file.tell() + extraLen

    @parseMaskData()
    @parseBlendingRanges()
    @parseLegacyLayerName()
    @parseLayerInfo()

    @file.seek @layerEnd
    return @
